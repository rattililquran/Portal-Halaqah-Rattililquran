  });

async function doLogin() {
  const email = document.getElementById('loginEmail').value.trim();
  if (!email) return showAuthError('Email wajib diisi');

  showLoading('Login...');
  try {
    const data = await Auth.login(email, 'guru');
    if (data.status !== 'ok') throw new Error(data.message || 'Login gagal');
    if (data.user.role !== 'guru') throw new Error('Akun ini bukan role Guru');

    token = data.token;
    currentUser = data.user;
    // Auth.login tidak simpan otomatis — simpan ke sessionStorage
    sessionStorage.setItem('hq_token', token);
    sessionStorage.setItem('hq_user', JSON.stringify(currentUser));
    startApp();
  } catch(e) {
    showAuthError(e.message);
  } finally {
    hideLoading();
  }
}

function doLogout() {
  Auth.logout(); // hapus token + redirect ke /index.html via api.js
  token = ''; currentUser = null; halaqahList = [];
}

  el.style.display = 'block';
}

async function bukaKBM() {
  const id_halaqah = document.getElementById('kbmHalaqahSelect').value;
  const tanggal = document.getElementById('kbmTanggal').value;
  const materi = document.getElementById('kbmMateri').value.trim();
  const metode = document.getElementById('kbmMetode').value;
  const catatan = document.getElementById('kbmCatatan').value.trim();

  if (!id_halaqah) return showToast('Pilih halaqah terlebih dahulu', 'error');
  if (!tanggal)    return showToast('Tanggal pertemuan wajib diisi', 'error');

  showLoading('Membuka sesi KBM...');
  try {
    const res = await GuruAPI.bukaKBM({
      id_halaqah, tanggal_pertemuan: tanggal,
      materi, metode, catatan_umum: catatan, durasi_menit: 60
    });
    sesiAktif = res.data;
    const halaqah = halaqahList.find(h => h.id_halaqah === id_halaqah);

    // Preload murid untuk sesi ini
    await loadMuridForKBM(id_halaqah);

    // Tampilkan sesi aktif
    document.getElementById('sesiAktifCard').style.display = 'block';
    document.getElementById('sesiAktifInfo').innerHTML =
      `<strong>${esc(halaqah?.nama_halaqah || id_halaqah)}</strong> &nbsp;·&nbsp; Pertemuan ke-${sesiAktif.pertemuan_ke} &nbsp;·&nbsp; ${formatDate(tanggal)} &nbsp;·&nbsp; <span class="badge badge-green">ID: ${esc(sesiAktif.id_kbm)}</span>`;

    showToast('Sesi KBM berhasil dibuka!', 'success');
  } catch(e) {
    showToast(e.message, 'error');
  } finally {
    hideLoading();
  }
}

async function tutupKBM() {
  if (!sesiAktif) return;
  if (!confirm('Tutup sesi KBM ini? Data tidak bisa diubah setelah ditutup.')) return;
  showLoading('Menutup sesi...');
  try {
    await GuruAPI.tutupKBM(sesiAktif.id_kbm);
    sesiAktif = null;
    document.getElementById('sesiAktifCard').style.display = 'none';
    showToast('Sesi KBM berhasil ditutup!', 'success');
    loadRiwayatKBM();
  } catch(e) {
    showToast(e.message, 'error');
  } finally {
    hideLoading();
  }
}

async function loadRiwayatKBM() {
  const id_halaqah = document.getElementById('riwayatHalaqahSelect').value;
  if (!id_halaqah) return;
  showLoading('Memuat riwayat...');
  try {
    const res = await GuruAPI.getRiwayatKBM(id_halaqah, 20);
    const rows = res.data || [];
    const tbody = document.getElementById('riwayatKBMTable');
    if (!rows.length) {
      tbody.innerHTML = `<tr><td colspan="6" style="text-align:center; color:var(--text-3); padding:24px;">Belum ada riwayat KBM</td></tr>`;
      return;
    }
    tbody.innerHTML = rows.map(k => `
      <tr>
        <td>${formatDate(k.tanggal_pertemuan)}</td>
        <td><span class="badge badge-blue">ke-${k.pertemuan_ke}</span></td>
        <td>${esc(k.materi || '-')}</td>
        <td><span class="badge badge-gray">${esc(k.metode || '-')}</span></td>
        <td><span style="color:var(--accent); font-weight:700;">${k.jumlah_hadir ?? 0}</span> hadir</td>
        <td>${statusBadge(k.status)}</td>
      </tr>`).join('');
  } catch(e) {
    showToast('Gagal memuat riwayat: ' + e.message, 'error');
  } finally {
    hideLoading();
  }
}

function statusBadge(s) {
  const map = { selesai:'badge-green', draft:'badge-warn', batal:'badge-red' };
  return `<span class="badge ${map[s]||'badge-gray'}">${s}</span>`;
}

async function loadMuridForKBM(id_halaqah) {
  if (muridCache[id_halaqah]) return muridCache[id_halaqah];
  const res = await GuruAPI.getMurid(id_halaqah);
  muridCache[id_halaqah] = res.data || [];
  return muridCache[id_halaqah];
}

async function loadMurid() {
  const id_halaqah = document.getElementById('muridHalaqahSelect').value;
  if (!id_halaqah) return;
  showLoading('Memuat murid...');
  try {
    const murid = await loadMuridForKBM(id_halaqah);
    const tbody = document.getElementById('muridTable');
    if (!murid.length) {
      tbody.innerHTML = `<tr><td colspan="7" style="text-align:center; color:var(--text-3); padding:24px;">Belum ada murid di halaqah ini</td></tr>`;
      return;
    }
    tbody.innerHTML = murid.map(m => {
      const hLast = m.hafalan_terakhir;
      return `<tr>
        <td><strong>${esc(m.nama_murid)}</strong></td>
        <td><small style="color:var(--text-3);">${esc(m.target_hafalan || '-')}</small></td>
        <td><span class="badge badge-green">${m.total_hadir ?? 0}x</span></td>
        <td>${m.total_setoran ?? 0}</td>
        <td>${m.total_murajaah ?? 0}</td>
        <td style="font-size:12px; color:var(--text-2);">
          ${hLast ? `${esc(hLast.surah_selesai)} : ${hLast.ayat_selesai} <span class="nilai-${hLast.nilai}">${hLast.nilai}</span>` : '<span style="color:var(--text-3);">–</span>'}
        </td>
        <td>
          <button class="btn btn-outline btn-sm" onclick="openCatatanModal('${esc(m.id_anggota)}','${esc(m.nama_murid)}','${esc(m.catatan_guru || '')}')">📝 Catatan</button>
        </td>
      </tr>`;
    }).join('');
  } catch(e) {
    showToast('Gagal memuat murid: ' + e.message, 'error');
  } finally {
    hideLoading();
  }
}

async function showAbsensiModal() {
  if (!sesiAktif) return showToast('Tidak ada sesi aktif', 'error');
  const murid = muridCache[sesiAktif.id_halaqah] || [];
  if (!murid.length) return showToast('Muat data murid terlebih dahulu', 'error');

  document.getElementById('absensiList').innerHTML = murid.map(m => `
    <div class="absensi-row">
      <div class="absensi-name">${esc(m.nama_murid)}</div>
      <div class="absensi-btns">
        ${['H','I','S','A'].map(k =>
          `<button class="btn-hadir ${k==='H'?'active':''}" data-kode="${k}" data-murid="${esc(m.id_murid)}"
            onclick="toggleAbsensi(this)">${k}</button>`
        ).join('')}
      </div>
    </div>`).join('');

  openModal('absensiModal');
}

function toggleAbsensi(btn) {
  const idMurid = btn.dataset.murid;
  const row = btn.closest('.absensi-btns');
  row.querySelectorAll('.btn-hadir').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
}

async function simpanAbsensi() {
  if (!sesiAktif) return;
  const absensi = [];
  document.querySelectorAll('#absensiList .absensi-row').forEach(row => {
    const aktif = row.querySelector('.btn-hadir.active');
    if (aktif) absensi.push({ id_murid: aktif.dataset.murid, status_hadir: aktif.dataset.kode });
  });
  if (!absensi.length) return showToast('Tidak ada data absensi', 'error');

  showLoading('Menyimpan absensi...');
  try {
    await GuruAPI.simpanAbsensi({
      id_kbm: sesiAktif.id_kbm,
      id_halaqah: sesiAktif.id_halaqah,
      absensi
    });
    closeModal('absensiModal');
    showToast(`Absensi ${absensi.length} murid berhasil disimpan!`, 'success');
  } catch(e) {
    showToast(e.message, 'error');
  } finally {
    hideLoading();
  }
}

function showHafalanModal() {
  if (!sesiAktif) return showToast('Tidak ada sesi aktif', 'error');
  const murid = muridCache[sesiAktif.id_halaqah] || [];
  const sel = document.getElementById('hafalanMuridSelect');
  sel.innerHTML = '<option value="">— Pilih Murid —</option>' +

}

async function simpanHafalan() {
  if (!sesiAktif) return;
  const id_murid = document.getElementById('hafalanMuridSelect').value;
  if (!id_murid) return showToast('Pilih murid terlebih dahulu', 'error');

  showLoading('Menyimpan hafalan...');
  try {
    await GuruAPI.simpanHafalan({
      id_kbm: sesiAktif.id_kbm,
      id_halaqah: sesiAktif.id_halaqah,
      id_murid,
      surah_mulai:   document.getElementById('hfSurahMulai').value,
      ayat_mulai:    document.getElementById('hfAyatMulai').value,
      surah_selesai: document.getElementById('hfSurahSelesai').value,
      ayat_selesai:  document.getElementById('hfAyatSelesai').value,
      jenis_hafalan: document.getElementById('hfJenis').value,
      nilai:         document.getElementById('hfNilai').value,
      lama_menit:    document.getElementById('hfLama').value,
      catatan_guru:  document.getElementById('hfCatatan').value,
    });
    closeModal('hafalanModal');
    showToast('Data hafalan berhasil disimpan!', 'success');
    // Reset form
    ['hfSurahMulai','hfAyatMulai','hfSurahSelesai','hfAyatSelesai','hfLama','hfCatatan'].forEach(id => {
      document.getElementById(id).value = '';
    });
    muridCache = {}; // invalidate cache
  } catch(e) {
    showToast(e.message, 'error');
  } finally {
    hideLoading();
  }
}

function openModal(id)  { document.getElementById(id).classList.add('open'); }
function closeModal(id) { document.getElementById(id).classList.remove('open'); }

function showLoading(msg = 'Memuat...') {
  document.getElementById('loadingText').textContent = msg;
  document.getElementById('loadingOverlay').classList.add('show');
}
function hideLoading() {
  document.getElementById('loadingOverlay').classList.remove('show');
}

let toastTimer;
function showToast(msg, type = '') {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.className = 'show ' + type;
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => el.className = '', 3000);
}

function formatDate(d) {
  if (!d) return '–';
  return new Date(d).toLocaleDateString('id-ID', { day:'numeric', month:'short', year:'numeric' });
}

function esc(str) {
  return String(str || '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}
