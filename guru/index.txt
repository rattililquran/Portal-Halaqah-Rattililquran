  });

async function doLogin() {
  const nis = document.getElementById('loginNIS').value.trim().toUpperCase();
  if (!nis) return showAuthError('NIS wajib diisi');

  setBtn('btnLogin', true, 'Masuk...');
  showLoading('Login...');
  try {
    const data = await Auth.login(nis, 'guru');
    if (data.status !== 'ok') throw new Error(data.message || 'Login gagal');
    if (data.user.role !== 'guru') throw new Error('Akun ini bukan role Guru');

    token = data.token;
    currentUser = data.user;
    sessionStorage.setItem('hq_token', token);
    sessionStorage.setItem('hq_user', JSON.stringify(currentUser));
    startApp();
  } catch(e) {
    showAuthError(e.message);
  } finally {
    hideLoading();
    setBtn('btnLogin', false, 'Masuk');
  }
}

function doLogout() {
  Auth.logout();
}

  el.style.display = 'block';
}

async function doBukaKBM() {
  const id_halaqah = document.getElementById('kbmHalaqahSelect').value;
  const tanggal    = document.getElementById('kbmTanggal').value;
  const materi     = document.getElementById('kbmMateri').value.trim();
  const metode     = document.getElementById('kbmMetode').value;
  const catatan    = document.getElementById('kbmCatatan').value.trim();

  if (!id_halaqah) return showToast('Pilih halaqah terlebih dahulu', 'error');
  if (!tanggal)    return showToast('Tanggal pertemuan wajib diisi', 'error');
  if (sesiAktif)   return showToast('Sudah ada sesi yang berjalan. Tutup dulu sebelum membuka baru.', 'warn');

  setBtn('btnBukaKBM', true, '⏳ Membuka...');
  showLoading('Membuka sesi KBM...');
  try {
    const res = await GuruAPI.bukaKBM({ id_halaqah, tanggal_pertemuan: tanggal, materi, metode, catatan_umum: catatan, durasi_menit: 60 });
    sesiAktif = res.data;

    // Preload murid untuk sesi ini
    await getMuridHalaqah(id_halaqah);
    showSesiAktif();
    showToast('Sesi KBM berhasil dibuka!', 'success');

    // Reset form
    document.getElementById('kbmMateri').value  = '';
    document.getElementById('kbmCatatan').value = '';
  } catch(e) {
    showToast(e.message, 'error');
  } finally {
    hideLoading();
    setBtn('btnBukaKBM', false, '▶ Buka Sesi KBM');
  }
}

async function doTutupKBM() {
  if (!sesiAktif) return;
  if (!confirm('Tutup sesi KBM ini? Data tidak bisa diubah setelah ditutup.')) return;

  showLoading('Menutup sesi...');
  try {
    const res = await GuruAPI.tutupKBM(sesiAktif.id_kbm);
    showToast('Sesi KBM berhasil ditutup! Hadir: ' + (res.data?.jumlah_hadir ?? 0) + ' murid.', 'success');
    sesiAktif = null;
    document.getElementById('sesiAktifBanner').style.display = 'none';
    // Refresh riwayat jika sedang di halaman itu
    if (document.getElementById('page-riwayat').classList.contains('active')) loadRiwayat();
  } catch(e) {
    showToast(e.message, 'error');
  } finally {
    hideLoading();
  }
}

async function doSimpanAbsensi() {

  });

  showLoading('Menyimpan absensi...');
  try {
    await GuruAPI.simpanAbsensi({ id_kbm: sesiAktif.id_kbm, id_halaqah: sesiAktif.id_halaqah, absensi });

    showToast(absensi.length + ' absensi berhasil disimpan!', 'success');

// ═══════════════════════════════════════════
//  MURID
// ═══════════════════════════════════════════
async function getMuridHalaqah(id_halaqah) {
  if (muridCache[id_halaqah]) return muridCache[id_halaqah];
  const res = await GuruAPI.getMurid(id_halaqah);
  muridCache[id_halaqah] = res.data || [];
  return muridCache[id_halaqah];
}

async function loadMurid() {
  const id_halaqah = document.getElementById('muridHalaqahSelect').value;
  if (!id_halaqah) return;
  showLoading('Memuat murid...');
  try {
    muridCache[id_halaqah] = null; // force refresh
    const murid = await getMuridHalaqah(id_halaqah);
    const tbody = document.getElementById('muridTable');
    if (!murid.length) {
      tbody.innerHTML = `<tr><td colspan="7" style="text-align:center;color:var(--text-3);padding:32px">Belum ada murid di halaqah ini</td></tr>`;
      return;
    }
    tbody.innerHTML = murid.map(m => {
      const hl = m.hafalan_terakhir;
      return `<tr>
        <td><strong>${esc(m.nama_murid)}</strong></td>
        <td><small style="color:var(--text-3)">${esc(m.target_hafalan||'-')}</small></td>
        <td><span class="badge badge-green">${m.total_hadir??0}×</span></td>
        <td>${m.total_setoran??0}</td>
        <td>${m.total_murajaah??0}</td>
        <td style="font-size:12px">
          ${hl ? `${esc(hl.surah_selesai)}:${hl.ayat_selesai} <span class="nilai-${hl.nilai}">${hl.nilai}</span>` : '<span style="color:var(--text-3)">–</span>'}
        </td>
        <td>
          <button class="btn btn-outline btn-sm" onclick="openCatatanModal('${esc(m.id_anggota)}','${esc(m.nama_murid)}','${esc((m.catatan_guru||'').replace(/'/g,"&#39;"))}')">📝 Catatan</button>
        </td>
      </tr>`;
    }).join('');
  } catch(e) {
    showToast('Gagal memuat murid: ' + e.message, 'error');
  } finally {
    hideLoading();
  }
}

  document.getElementById('catatanNama').textContent      = nama;
  document.getElementById('catatanInput').value           = catatan;
  document.getElementById('catatanIdAnggota').value       = id_anggota;
  openModal('catatanModal');
}

async function doSimpanCatatan() {
  const id_anggota  = document.getElementById('catatanIdAnggota').value;
  const catatan_guru = document.getElementById('catatanInput').value;
  showLoading('Menyimpan catatan...');
  try {
    await GuruAPI.updateCatatanMurid({ id_anggota, catatan_guru });
    closeModal('catatanModal');
    // Update cache
    const select = document.getElementById('muridHalaqahSelect');
    if (select.value) muridCache[select.value] = null;
    showToast('Catatan berhasil disimpan!', 'success');
  } catch(e) {
    showToast(e.message, 'error');
  } finally {
    hideLoading();
  }
}

async function loadRiwayat() {
  const id_halaqah = document.getElementById('riwayatHalaqahSelect').value;
  if (!id_halaqah) return;
  showLoading('Memuat riwayat...');
  try {
    const res  = await GuruAPI.getRiwayatKBM(id_halaqah, 30);
    const rows = res.data || [];
    const tbody = document.getElementById('riwayatTable');
    if (!rows.length) {
      tbody.innerHTML = `<tr><td colspan="7" style="text-align:center;color:var(--text-3);padding:32px">Belum ada riwayat KBM</td></tr>`;
      return;
    }
    tbody.innerHTML = rows.map(k => `<tr>
      <td>${formatDate(k.tanggal_pertemuan)}</td>
      <td><span class="badge badge-blue">ke-${k.pertemuan_ke}</span></td>
      <td>${esc(k.materi||'-')}</td>
      <td><span class="badge badge-gray">${esc(k.metode||'-')}</span></td>
      <td><strong style="color:var(--accent)">${k.jumlah_hadir??0}</strong> hadir / <span style="color:var(--danger)">${k.jumlah_alpa??0}</span> alpa</td>
      <td>${statusBadge(k.status)}</td>
      <td><button class="btn btn-outline btn-sm" onclick="loadDetailKBM('${esc(k.id_kbm)}','${esc(k.tanggal_pertemuan)}')">🔍 Detail</button></td>
    </tr>`).join('');
  } catch(e) {
    showToast('Gagal memuat riwayat: ' + e.message, 'error');
  } finally {
    hideLoading();
  }
}

async function loadDetailKBM(id_kbm, tanggal) {
  document.getElementById('detailKBMTitle').textContent = 'Detail Hafalan — ' + formatDate(tanggal);
  document.getElementById('detailKBMBody').innerHTML =
    '<div class="empty-state"><div class="empty-icon">⏳</div><div class="empty-title">Memuat...</div></div>';
  openModal('detailKBMModal');

  try {
    const res  = await GuruAPI.getHafalanByKBM(id_kbm);
    const data = res.data || [];
    if (!data.length) {
      document.getElementById('detailKBMBody').innerHTML =
        '<div class="empty-state"><div class="empty-icon">📭</div><div class="empty-title">Belum ada data hafalan untuk sesi ini</div></div>';
      return;
    }
    document.getElementById('detailKBMBody').innerHTML = data.map(hf => `
      <div class="hafalan-row">
        <div style="display:flex;align-items:center;justify-content:space-between;margin-bottom:6px">
          <strong>${esc(hf.id_murid)}</strong>
          <div style="display:flex;gap:6px;align-items:center">
            ${statusHadirBadge(hf.status_hadir)}
            ${hf.nilai ? `<span class="nilai-${hf.nilai}">${hf.nilai}</span>` : ''}
            ${hf.jenis_hafalan ? `<span class="badge badge-blue">${esc(hf.jenis_hafalan)}</span>` : ''}
          </div>
        </div>
        <div style="font-size:12px;color:var(--text-2)">
          ${hf.surah_mulai ? `📖 ${esc(hf.surah_mulai)} ${hf.ayat_mulai} — ${esc(hf.surah_selesai)} ${hf.ayat_selesai}` : '<em style="color:var(--text-3)">Tidak ada setoran</em>'}
          ${hf.lama_menit ? ` &nbsp;·&nbsp; ⏱ ${hf.lama_menit} menit` : ''}
        </div>
        ${hf.catatan_guru ? `<div style="font-size:12px;color:var(--text-3);margin-top:4px">💬 ${esc(hf.catatan_guru)}</div>` : ''}
      </div>`).join('');
  } catch(e) {
    document.getElementById('detailKBMBody').innerHTML =
      `<div class="empty-state"><div class="empty-icon">❌</div><div class="empty-title">Gagal memuat: ${esc(e.message)}</div></div>`;
  }
}

function openModal(id)  { document.getElementById(id).classList.add('open') }
function closeModal(id) { document.getElementById(id).classList.remove('open') }

function showLoading(msg = 'Memuat...') {
  document.getElementById('loadingText').textContent = msg;
  document.getElementById('loadingOverlay').classList.add('show');
}
function hideLoading() { document.getElementById('loadingOverlay').classList.remove('show') }

let _toastTimer;
function showToast(msg, type = '') {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.className   = 'show ' + type;
  clearTimeout(_toastTimer);
  _toastTimer = setTimeout(() => { el.className = '' }, 3500);
}

function setBtn(id, disabled, text) {
  const el = document.getElementById(id);
  if (!el) return;
  el.disabled   = disabled;
  el.textContent = text;
}

function statusBadge(s) {
  const map = { selesai:'badge-green', draft:'badge-warn', batal:'badge-red' };
  return `<span class="badge ${map[s]||'badge-gray'}">${s||'-'}</span>`;
}

function statusHadirBadge(k) {
  const map = { H:'badge-green', I:'badge-warn', S:'badge-blue', A:'badge-red' };
  const lbl = { H:'Hadir', I:'Izin', S:'Sakit', A:'Alpa' };
  return `<span class="badge ${map[k]||'badge-gray'}">${lbl[k]||k||'-'}</span>`;
}

function formatDate(d) {
  if (!d) return '–';
  return new Date(d).toLocaleDateString('id-ID', { day:'numeric', month:'short', year:'numeric' });
}

function esc(str) {
  return String(str ?? '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function emptyState(icon, title, sub = '') {
  return `<div class="empty-state"><div class="empty-icon">${icon}</div><div class="empty-title">${title}</div>${sub ? `<div class="empty-sub">${sub}</div>` : ''}</div>`;
}
